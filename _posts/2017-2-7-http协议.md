**利用HTTP协议采集网站信息**

    echo "<pre>";
    //fsockopen（地址，端口，错误码，错误信息，超时)返回资源类型
    $fh = fsockopen ('192.168.1.113' , 80 ,$errno , $errstr , 5);
    //
    $req = array(
     'GET /http/05.html HTTP/1.1',
     'Host: localhost',
     '',
     ''
    );
    //拆分请求
    $req = implode("\n", $req);
    //斜土资源
    fwrite($fh, $req);
    //循环读取返回值
    while($row = fread($fh,32)) {
     echo $row;
    }
    fclose($fh)
