## 图片自动切换类似淘宝首页广告版 ##

    <!DOCTYPE html>
    <html>
    <head>
    	<title></title>
    	<style type="text/css">
    	div{
    		width: 150px;
    		height:180px;
    		overflow: hidden;
    		position: relative;
    	}
    
    	ul{
    		position: absolute;
    		bottom: 5px;
    		right: 5px;
    		width: 50px;
    		height: 20px;'
    		list-style:none;	
    		}
    
    	li{
    		float: left;
    		width: 20px;
    		height: 10px;
    		background-color: gray;
    		border-radius: 100%;
    	}
    
    	.active{
    		background-color: red;
    	}
    	</style>
    </head>
    <body>
    <div id="ad">
    	<img src="on.jpg" />
    	<img src="off.jpg" />
    	<img src="on.jpg" />
    	<ul>
    		<li class="active">1</li>
    		<li>2</li>
    		<li>3</li>
    	</ul>
    </div>
    
    </body>
    <script src="jquery.js"></script>
    <script type="text/javascript">
    var i=0;
    setInterval(function (){
    	$('#ad img:first').fadeOut(2000,function(){
    		$(this).appendTo('#ad').fadeIn();
    		$('li').removeClass('active');
    		$('li').get(++i%3).className = 'active';
    	});
    
    },2000);
    </script>
